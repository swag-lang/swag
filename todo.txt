#test
{
    {
        let x1 = 10's32
        let x2 = 20's32
        var s: s32

        s = x1 <=> x2
        @assert(s == -1)
        s = x2 <=> x1
        @assert(s == 1)
        s = x1 <=> x1
        @assert(s == 0)
    }

    // fail!
    {
        let x1 = 10'u32
        let x2 = 20'u32
        var s: s32

        s = x1 <=> x2
        @assert(s == -1)
        s = x2 <=> x1
        @assert(s == 1)
        s = x1 <=> x1
        @assert(s == 0)
    }
}
